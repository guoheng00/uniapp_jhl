    # Form 组件

    该组件内部实现以 uni-ui 组件为基础，进行二次封装

    # 该组件主要应用场景 :

    • 表单

    WARNING

    注意

    • 此组件内部使用 uni-ui 组件为基础，详细说明请参 uni-ui 官方文档

    # 基础使用

    通过<g-form>形式来调用

      <template>
	<view class="g-form">
		<g-form :data="data" ref="dynamicForm" :formDefinition="formDefinition" @handlerClickInfo="handlerClickInfo"
			@handleSelectClick="handleSelectClick">

			<!-- <uni-forms-item label='' label-width="120" :rules="[{'required': false,errorMessage: '域名项必填13123'}]"
				name='zdy'>

				<up-input v-model="data.zdy"></up-input>
			</uni-forms-item>-->
			<template #zdy>
				<g-input v-model="data.zdy1"></g-input>
			</template>
		</g-form>
		<g-button type="primary" text="提交" @handleButtonClick="handleSubmit"></g-button>
	</view>

</template>

	import { ref, reactive, onMounted } from 'vue';
	const dynamicForm = ref(null)
	// 响应式表单数据  
	const data = reactive<any>({
		text: '', // 注意:这里的对象key要跟表单定义对象中的name属性值一一对应

		select: '',

		zdy: "",
		zdy1: "",

			labelAlign: 'left',  //label 对齐方式  left  center right

				name: 'text', // 每个表单项的唯一标识(不能重复)
				type: 'input', // 表单项的类型 这里是输入框
				inputType: 'text', // 输入框的类型,有 text number digit textarea
				password: false, // 是否为密码输入框
				placeholder: '请输入您的姓名', // 提示信息
				disabled: false, // 是否禁用
				maxLength: -1,// 最大输入长度
				inputAligin: 'left',//输入框文字位置 
				labelWidth: 75,// label的宽度  单位px   来调节跟详情图标的距离
				inputInfo: false,//详情说明按钮图标
				required: true,//是否带红色 **
				// 校验规则
				rules: [
					{ 'required': true, errorMessage: '请输入您的姓名' }
				]
			},
			// {
			// 	label: 'number 输入', // 表单显示名称

			// 	placeholder: '请选择', // 提示信息
			// 	disabled: false, // 是否禁用
			// 	maxLength: -1,// 最大输入长度
			// 	inputAligin: 'right',//输入框文字位置
			// 	labelWidth: 140// label的宽度  单位px
			// },
			// {

			// 	name: 'digit', // 每个表单项的唯一标识(不能重复)

			// 	password: false, // 是否为密码输入框

			// 	maxLength: -1,// 最大输入长度

			// 	labelWidth: 110// label的宽度  单位px
			// },
			// {

			// 	name: 'textarea', // 每个表单项的唯一标识(不能重复)

						text: '不知道',

		// console.log(dynamicForm.value.dynamicForm)

    # Api

    # Props

    属性名 说明 类型 默认值 可选值 平台差异说明
    data 表单绑定的值 object -
    formDefinition 配置项 object -

    # forms 配置项

    属性名 说明 类型 默认值 可选值 平台差异说明
    labelPosition label 的位置 top left string -
    labelAlign label 对齐方式 输入框有效 string left center right -

    # fields 配置项

    属性名 说明 类型 默认值 可选值 平台差异说明
    label 表单 label string -
    name form-items 绑定数据 string -
    type form-items 类型 例如 input select switch slot string input select switch slot date radio -
    inputType type=input 下有效 string text number digit textarea -
    password type=input 下有效 是否为密码框 boolean -
    placeholder 表单提示语 string -
    disabled 是否禁用 boolean -
    maxLength 最大输入长度 -1 不限制 number -
    inputAligin 输入框文字位置 string -
    labelWidth label 的宽度 单位 px 来调节跟详情图标的距离 number -
    inputInfo 详情说明按钮图标是否显示 boolean -
    required 是否带红色 * boolean -
    rules 校验规则表单 object -
    iconUrl 选择框右侧 icon 图标的链接 type=select 下有效 string -
    activeValue 打开的时候绑定的值 type=switch 下有效 object -
    inactiveValue 关闭的时候绑定的值 type=switch 下有效 object -
    space 圆点的边距 type=switch 下有效 object -
    size 开关的大小 type=switch 下有效 object -
    hobbys 复选框绑定的值 type=date 下有效 array -
    options 单选选项 type=radio 下有效 array -
    slotName 单选选项 type=slot 下有效 具名插槽 array -

    # Event

    属性名 说明 类型 默认值 可选值 平台差异说明
    handlerClickInfo 点击 label 附近的 信息图标按钮事件 - - -
    handleSelectClick 选择框的点击事件 根据 type 来分辨类型 比如 select 下拉 date 时间 - -
