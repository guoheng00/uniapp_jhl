    # pupop-list 组件

    该组件内部实现以 uview-plus 组件为基础，进行二次封装

    # 该组件主要应用场景 :

    • 弹窗列表搜索

    WARNING

    注意

    • 此组件内部使用 uview-plus uni-ui 组件为基础，详细说明请参 uview-plus 官方文档

    # 基础使用

    通过<g-popup-list>形式来调用

    <template>
	<view class="g-popup-list">
		<g-button type="primary" text="打开搜索列表" @handleButtonClick="handleBtnClick"></g-button>
		<view class="">
			选中了列表: {{listName}}
		</view>
		<g-popup-list :show="show" title="传入标题" searchPlaceholder="请输入关键字搜索" :optionData="optionData"
			:loadingStatus='loadingStatus' :loadmoreText="loadmoreText" :loadingText="loadingText"
			:nomoreText="nomoreText" :maxPageNo="pageData.maxPageNo" :triggered="triggered" @handleClose="handleClose"
			@handleOpen="handleOpen" @handleConfirm="handleConfirm" @handleSearch="handleSearch"
			@handleClear="handleClear" @lower="handleLower" @refresherrefresh="handleRefresherrefresh">
		</g-popup-list>
	</view>
</template>

<script lang="ts" setup>
	import { ref } from 'vue';
	const listName = ref('')
	// 创建响应式数据
	const show = ref(true);
	//列表数据
	const optionData = ref([
		{
			label: "列表1",
			id: 1
		},
		{
			label: "列表2",
			id: 2
		},
		{
			label: "列表3",
			id: 3
		},
		{
			label: "列表4",
			id: 4
		},
		{
			label: "列表5",
			id: 5
		},
		{
			label: "列表6",
			id: 6
		},
		{
			label: "列表7",
			id: 7
		},
		{
			label: "列表8",
			id: 8
		},
		{
			label: "列表9",
			id: 9
		},
	])
	//分页数据
	const pageData = ref({
		pageNo: 1,
		pageSize: 10,
		maxPageNo: 2
	})
	//下拉刷新开关
	const triggered = ref(false)
	// 底部加载状态
	const loadingStatus = ref('loadmore');
	const loadmoreText = ref('轻轻上拉');
	const loadingText = ref('努力加载中');
	const nomoreText = ref('实在没有了');
	const handleBtnClick = () => {
		show.value = true
	}
	// 定义方法
	function handleOpen() {
		// 打开逻辑，比如设置 show 为 true
	}

	function handleClose() {
		// 关闭逻辑，设置 show 为 false
		show.value = false;
	}
	// 搜索功能
	const handleSearch = (value : number | string) => {
		// console.log(value);
		optionData.value = [{
			label: '列表10',
			id: 10
		}]
	}
	//  清除搜索框 handleClear 事件
	const handleClear = (value : number | string) => {
		// console.log(value);
		optionData.value = [{
			label: '列表11',
			id: 11
		}]
	}
	// 下拉刷新  上拉加载更多功能
	const handleRefresherrefresh = () => {
		// console.log('下拉刷新');
		triggered.value = true
		setTimeout(() => {
			triggered.value = false
		}, 2000)
	}
	const handleLower = () => {
		// console.log('上拉加载更多,到达底部');
		loadingStatus.value = 'loading'
	}
	// 确定功能  选择值
	const handleConfirm = (active : number | string) => {
		if (active) {
			const foundObject : any = optionData.value.find(item => item.id === active);
			listName.value = foundObject.label
		}
		show.value = false;
	}
</script>

    # Api

    # Props

    参数 说明 类型 默认值 可选值
    show 是否显示弹窗，请赋值给 show Boolean false true
    title 传入标题 String - -
    searchPlaceholder 传入列表搜索提示词 String - -
    loadingStatus 传入上拉加载状态 String - loadmore- loading -nomore
    loadmoreText 加载提示词 String - -
    loadingText 加载中的文字 String - -
    nomoreText 加载结束文字 String - -
    triggered 下拉刷新开关 Boolean - -
    maxPageNo 数据最大页数 Number - 最大页数小于 2 则不显示底部加载提示
    delay 节流事件 Number - 毫秒

    # Event

    事件名 说明 回调参数
    handleClose 弹窗关闭事件 -
    handleOpen 弹窗打开事件 -
    handleConfirm 列表确认按钮事件 -
    handleSearch 搜索框回车 手机键盘点击搜索事件 -
    handleClear 搜索框清空事件 -
    handleLower 滚动到底部事件,节流事件 600 -
    handleRefresherrefresh 下拉刷新事件 -
